50  // Encode a sequence of movements as an unsigned integer.
51  unsigned int EncodeTrajectory(const std::vector<Movement2D>& movements) {
52  const unsigned int base =
53  Movement2D::GetNumDeltaXs() * Movement2D::GetNumDeltaYs() *
54  Movement2D::GetNumDeltaAngles();
55 
56  unsigned int id = 0;
57  unsigned int place_value = 1;
58 
59  for (size_t ii = 0; ii < movements.size(); ii++) {
60  const unsigned int x_id = movements[ii].GetIndexX();
61  const unsigned int y_id = movements[ii].GetIndexY();
62  const unsigned int a_id = movements[ii].GetIndexAngle();
63 
64  // Compute a unique identifier for this movement.
65  const unsigned int step_id =
66  x_id + y_id * Movement2D::GetNumDeltaXs() +
67  a_id * Movement2D::GetNumDeltaXs() * Movement2D::GetNumDeltaYs();
68 
69  // Update the overall trajectory id, and the place value.
70  id += step_id * place_value;
71  place_value *= base;
72  }
73 
74  return id;
75  }

77  // Decode a trajectory id into a sequence of poses.
78  void DecodeTrajectory(unsigned int id, unsigned int num_steps,
79  const GridPose2D& initial_pose,
80  std::vector<GridPose2D>& trajectory) {
81  trajectory.clear();
82  const unsigned int base =
83  Movement2D::GetNumDeltaXs() * Movement2D::GetNumDeltaYs() *
84  Movement2D::GetNumDeltaAngles();
85 
86  GridPose2D current_pose = initial_pose;
87  while (id > 0) {
88  const unsigned int remainder = id % base;
89 
90  // Convert remainder to Movement2D.
91  const unsigned int x_id = remainder % Movement2D::GetNumDeltaXs();
92  const unsigned int y_id =
93  (remainder / Movement2D::GetNumDeltaXs()) % Movement2D::GetNumDeltaYs();
94  const unsigned int a_id =
95  remainder / (Movement2D::GetNumDeltaXs() * Movement2D::GetNumDeltaYs());
96 
97  const Movement2D step(x_id, y_id, a_id);
98 
99  // Append to trajectory.
100  GridPose2D next_pose = current_pose;
101  CHECK(next_pose.MoveBy(step));
102  trajectory.push_back(next_pose);
103 
104  // Update 'id'.
105  id /= base;
106 
107  // Reset 'current_pose'.
108  current_pose = next_pose;
109  }
110 
111  // If the above does not recover enough poses, the last remainders were zero.
112  // Update 'trajectory' accordingly.
113  while (trajectory.size() < num_steps) {
114  GridPose2D next_pose = current_pose;
115  CHECK(next_pose.MoveBy(Movement2D(0, 0, 0)));
116 
117  trajectory.push_back(next_pose);
118  current_pose = next_pose;
119  }
120  }

122  // Encode a sequence of measurements in an unsigned integer.
123  unsigned int EncodeMeasurements(const std::vector<unsigned int>& measurements,
124  unsigned int max_measurement) {
125  const unsigned int base = max_measurement + 1;
126 
127  unsigned int id = 0;
128  unsigned int place_value = 1;
129 
130  for (size_t ii = 0; ii < measurements.size(); ii++) {
131  id += measurements[ii] * place_value;
132  place_value *= base;
133  }
134 
135  return id;
136  }

138  // Decode a measurement id into a list of measurements.
139  void DecodeMeasurements(unsigned int id, unsigned int max_measurement,
140  unsigned int num_measurements,
141  std::vector<unsigned int>& measurements) {
142  measurements.clear();
143  const unsigned int base = max_measurement + 1;
144 
145  while (id > 0) {
146  const unsigned int remainder = id % base;
147  measurements.push_back(remainder);
148 
149  // Update 'id'.
150  id /= base;
151  }
152 
153  // If not enough measurements, the rest must have been zero.
154  while (measurements.size() < num_measurements) {
155  measurements.push_back(0);
156  }
157  }

159  // Encode a list of sources (map) as an unsigned integer.
160  unsigned int EncodeMap(const std::vector<Source2D>& sources,
161  unsigned int num_rows, unsigned int num_cols) {
162  const unsigned int base = num_rows * num_cols;
163 
164  unsigned int id = 0;
165  unsigned int place_value = 1;
166 
167  for (size_t ii = 0; ii < sources.size(); ii++) {
168  const unsigned int source_id =
169  sources[ii].GetIndexX() + sources[ii].GetIndexY() * num_rows;
170 
171  id += source_id * place_value;
172  place_value *= base;
173  }
174 
175  return id;
176  }

178  // Decode a map id into a list of sources.
179  void DecodeMap(unsigned int id, unsigned int num_rows, unsigned int num_cols,
180  unsigned int num_sources, std::vector<Source2D>& sources) {
181  sources.clear();
182  const unsigned int base = num_rows * num_cols;
183 
184  while (id > 0) {
185  const unsigned int remainder = id % base;
186 
187  // Unpack remainder into a source.
188  const unsigned int x_id = remainder % num_rows;
189  const unsigned int y_id = remainder / num_rows;
190  sources.push_back(Source2D(x_id, y_id));
191 
192  // Update 'id'.
193  id /= base;
194  }
195 
196  // If not enough sources, the remainders must have been zero.
197  const unsigned int zero = 0;
198  while (sources.size() < num_sources) {
199  sources.push_back(Source2D(zero, zero));
200  }
201  }
