38 //
39 // This file defines cost functors that can be used in conjunction with Google
40 // Ceres solver to solve non-linear least-squares problems. Each functor must
41 // define a public templated operator() method that takes in arguments const T*
42 // const INPUT_VARIABLE, and T* OUTPUT_RESIDUAL, and returns a bool.
43 // 'INPUT_VARIABLE' will be the optimization variable, and 'OUTPUT_RESIDUAL'
44 // will be the cost associated with the input.
45 //
46 // One can define more specific cost functions by adding other structure to the
47 // functor, e.g. by passing in other parameters of the cost function to the
48 // functor's constructor.
49 //
50 ///////////////////////////////////////////////////////////////////////////////

63 // Belief error measures the difference between the expected sensor measurement
64 // and the true measurement, assuming the given belief vector of probabilities
65 // that each voxel is occupied.
66 struct BeliefError {
67  // Inputs: pointer to list of voxel indices that are in view, and measurement.
68  // Optimization variables are the probability that each voxel contains a
69  // source.
70  const std::vector<unsigned int>* voxels_;
71  const unsigned int measurement_;
72 
73  BeliefError(const std::vector<unsigned int>* voxels,
74  unsigned int measurement)
75  : voxels_(voxels), measurement_(measurement) {
76  CHECK_NOTNULL(voxels);
77  }
78 
79  template <typename T>
80  bool operator()(T const* const* belief, T* expected_error) const {
81  // Compute the difference between the expected measurement and the
82  // actual measurement.
83  *expected_error = -static_cast<T>(measurement_);
84 
85  for (const auto& voxel_id : *voxels_) {
86  *expected_error += belief[0][voxel_id];
87  }
88 
89  return true;
90  }
91 
92  // Factory method.
93  static ceres::CostFunction* Create(unsigned int num_rows,
94  unsigned int num_cols,
95  const std::vector<unsigned int>* voxels,
96  const unsigned int measurement) {
97  // Only a single residual.
98  const int kNumResiduals = 1;
99 
100  // Number of parameters is the number of grid cells.
101  const int kNumParameters = static_cast<int>(num_rows * num_cols);
102 
103  // Stride. Number of derivatives to calculate. See below for details:
104  // http://ceres-solver.org/nnls_modeling.html#dynamicautodiffcostfunction
105  const int kStride = 4;
106 
107  ceres::DynamicAutoDiffCostFunction<BeliefError, kStride>* cost =
108  new ceres::DynamicAutoDiffCostFunction<BeliefError, kStride>(
109  new BeliefError(voxels, measurement));
110  cost->AddParameterBlock(kNumParameters);
111  cost->SetNumResiduals(kNumResiduals);
112  return cost;
113  }
114 }; // struct BeliefError

116 // Belief regularization enforces consistency across the entire belief vector
117 // by measuring the difference between the expected number of sources and the
118 // total number that should be present.
119 struct BeliefRegularization {
120  // Inputs: true number of sources and regularization tradeoff parameter.
121  // Optimization variables are the probability that each voxel contains a
122  // source.
123  const unsigned int num_parameters_;
124  const unsigned int num_sources_;
125  const double regularizer_;
126 
127  BeliefRegularization(unsigned int num_parameters, int num_sources,
128  double regularizer)
129  : num_parameters_(num_parameters),
130  num_sources_(num_sources),
131  regularizer_(sqrt(regularizer)) {}
132 
133  template <typename T>
134  bool operator()(T const* const* belief, T* expected_error) const {
135  // Compute the difference between the expected measurement and the
136  // actual measurement.
137  *expected_error = -static_cast<T>(num_sources_);
138 
139  for (size_t ii = 0; ii < num_parameters_; ii++) {
140  *expected_error += belief[0][ii];
141  }
142 
143  *expected_error *= static_cast<T>(regularizer_);
144 
145  return true;
146  }
147 
148  // Factory method.
149  static ceres::CostFunction* Create(unsigned int num_rows,
150  unsigned int num_cols,
151  unsigned int num_sources,
152  double regularizer) {
153  // Only a single residual.
154  const int kNumResiduals = 1;
155 
156  // Number of parameters is the number of grid cells.
157  const int kNumParameters = static_cast<int>(num_rows * num_cols);
158 
159  // Stride. Number of derivatives to calculate. See below for details:
160  // http://ceres-solver.org/nnls_modeling.html#dynamicautodiffcostfunction
161  const int kStride = 4;
162 
163  ceres::DynamicAutoDiffCostFunction<BeliefRegularization, kStride>* cost =
164  new ceres::DynamicAutoDiffCostFunction<BeliefRegularization, kStride>(
165  new BeliefRegularization(num_rows * num_cols,
166  num_sources, regularizer));
167 
168  cost->AddParameterBlock(kNumParameters);
169  cost->SetNumResiduals(kNumResiduals);
170  return cost;
171  }
172 }; // struct BeliefRegularization
