71  // Generate random sources according to the current belief state.
72  bool GridMap2D::GenerateSources(std::vector<Source2D>& sources) {
73  const double total_belief = belief_.sum();
74  std::uniform_real_distribution<double> unif(0.0, 1.0);
75 
76  // Choose 'num_sources_' random numbers in [0, 1], which will be sorted
77  // and treated as evaluations of the CDF. Since they are uniform, the points
78  // at which they occur ar distributed according to the current belief.
79  std::vector<double> cdf_evals;
80  for (size_t ii = 0; ii < num_sources_; ii++)
81  cdf_evals.push_back(unif(rng_));
82 
83  std::sort(cdf_evals.begin(), cdf_evals.end());
84 
85  // Walk the current belief distribution until we get to each 'cdf_eval'
86  // and record which voxel we are in.
87  sources.clear();
88  unsigned int current_index = 0;
89  double current_cdf = 0.0;
90 
91  for (unsigned int ii = 0; ii < num_rows_; ii++) {
92  for (unsigned int jj = 0; jj < num_cols_; jj++) {
93  current_cdf += belief_(ii, jj) / total_belief;
94 
95  // Check if we just passed the next 'cdf_eval'.
96  while (current_cdf >= cdf_evals[current_index]) {
97  // Generate a new source here.
98  sources.push_back(Source2D(ii, jj));
99 
100  // Return if we have enough sources.
101  if (sources.size() == num_sources_)
102  return true;
103 
104  // Increment 'current_index'.
105  current_index++;
106  }
107  }
108  }
109 
110  // Should never get here.
111  return false;
112  }

114  // Generate entropy vector [h_{Z|X}], where the i-entry of [h_{Z|X}]
115  // is the entropy of Z given trajectory X = i, starting from the given pose.
116  void GridMap2D::GenerateEntropyVector(
117  unsigned int num_samples, unsigned int num_steps, const GridPose2D& pose,
118  double sensor_fov, Eigen::VectorXd& hzx,
119  std::vector<unsigned int>& trajectory_ids) {
120  // Compute the number of possible measurement vectors.
121  const unsigned int kNumMeasurements = pow(num_sources_ + 1, num_steps);
122 
123  // Create a map to keep track of counts for each trajectory.
124  std::map<unsigned int, Eigen::VectorXd> zx_samples;
125 
126  // Generate a ton of sampled data.
127  for (unsigned int ii = 0; ii < num_samples; ii++) {
128  // Generate random sources on the grid according to the current 'belief',
129  // and compute a corresponding 'map_id' number based on which grid cells
130  // the sources lie in.
131  std::vector<Source2D> sources;
132  if (!GenerateSources(sources)) {
133  VLOG(1) << "Unable to generate sources. Skipping this sample.";
134  continue;
135  }
136 
137  // Pick a random trajectory starting at the given pose. At each step,
138  // take a measurement and record the data.
139  GridPose2D current_pose = pose;
140  std::vector<Movement2D> movements;
141  std::vector<unsigned int> measurements;
142  while (movements.size() < num_steps) {
143  const Movement2D step;
144  if (current_pose.MoveBy(step)) {
145  movements.push_back(step);
146 
147  const Sensor2D sensor(current_pose, sensor_fov);
148  measurements.push_back(sensor.Sense(sources));
149  }
150  }
151 
152  // Compute trajectory and measurement sequence ids.
153  const unsigned int trajectory_id = EncodeTrajectory(movements);
154  const unsigned int measurement_id =
155  EncodeMeasurements(measurements, num_sources_);
156 
157  // Record this sample in the 'zx_samples' map.
158  if (zx_samples.count(trajectory_id) == 0) {
159  Eigen::VectorXd counts = Eigen::VectorXd::Zero(kNumMeasurements);
160  counts(measurement_id) = 1.0;
161  zx_samples.insert({trajectory_id, counts});
162  } else {
163  zx_samples.at(trajectory_id)(measurement_id) += 1.0;
164  }
165  }
166 
167  // Convert 'zx_samples' into a matrix joint distribution.
168  const unsigned int kNumTrajectories = zx_samples.size();
169  Eigen::MatrixXd pzx =
170  Eigen::MatrixXd::Zero(kNumMeasurements, kNumTrajectories);
171  trajectory_ids.clear();
172  trajectory_ids.reserve(kNumTrajectories);
173 
174  unsigned int idx = 0;
175  for (auto pair = zx_samples.begin(); pair != zx_samples.end(); pair++) {
176  trajectory_ids.push_back(pair->first);
177  pzx.col(idx++) = pair->second;
178  }
179 
180  // Normalize so that all columns sum to unity.
181  for (unsigned int jj = 0; jj < kNumTrajectories; jj++) {
182  const double zx_col_sum = pzx.col(jj).sum();
183  if (zx_col_sum < 1.0) {
184  VLOG(1) << "Encountered trajectory with no support in P_{Z|X}.";
185  } else {
186  pzx.col(jj) /= zx_col_sum;
187  }
188  }
189 
190  // Compute [h_{Z|X}], the conditional entropy vector.
191  hzx.resize(kNumTrajectories);
192  for (unsigned int jj = 0; jj < kNumTrajectories; jj++) {
193  hzx(jj) = 0.0;
194 
195  for (unsigned int ii = 0; ii < kNumMeasurements; ii++) {
196  const double p = pzx(ii, jj);
197 
198  // Catch 'p' values near 0 or 1 to avoid numerical issues.
199  if (p < 0.01 || p > 1.0 - 0.01)
200  continue;
201 
202  hzx(jj) -= p * log(p);
203  }
204 
205  // Make sure entropies are non-negative.
206  CHECK(hzx(jj) >= 0.0);
207  }
208  }

210  // Take a measurement from the given sensor and update belief accordingly.
211  bool GridMap2D::Update(const Sensor2D& sensor,
212  const std::vector<Source2D>& sources,
213  bool solve) {
214  const unsigned int measurement = sensor.Sense(sources);
215  CHECK(measurement <= num_sources_);
216 
217  // Identify all voxels in range and store.
218  std::vector<unsigned int> voxels;
219  for (size_t ii = 0; ii < num_rows_; ii++) {
220  for (size_t jj = 0; jj < num_cols_; jj++) {
221  if (sensor.VoxelInView(ii, jj))
222  voxels.push_back(ii + jj * num_rows_);
223  }
224  }
225 
226  viewed_.push_back(voxels);
227  measurements_.push_back(measurement);
228 
229  // Maybe solve.
230  if (solve)
231  return SolveLeastSquares();
232 
233  return true;
234  }

236  // Compute entropy.
237  double GridMap2D::Entropy() const {
238  double entropy = 0.0;
239 
240  for (size_t ii = 0; ii < num_rows_; ii++) {
241  for (size_t jj = 0; jj < num_cols_; jj++) {
242  const double p = belief_(ii, jj);
243  const double bernoulli_entropy = (p > 1e-8 && p < 1.0 - 1e-8) ?
244  (-p * log(p) - (1.0 - p) * log(1.0 - p)) : 0.0;
245 
246  entropy += bernoulli_entropy;
247  }
248  }
249 
250  return entropy;
251  }

253  // Solve least squares problem to update belief state.
254  bool GridMap2D::SolveLeastSquares() {
255  // Create a non-linear least squares problem.
256  ceres::Problem problem;
257 
258  // Add residual blocks for each set of observed voxels and their
259  // associated measurements. Note that 'belief_' is laid out in column-
260  // major order by default, so we can just use its underlying structure
261  // as the optimization variable.
262  for (size_t ii = 0; ii < viewed_.size(); ii++) {
263  problem.AddResidualBlock(
264  BeliefError::Create(num_rows_, num_cols_,
265  &viewed_[ii], measurements_[ii]),
266  NULL, /* squared loss */
267  belief_.data());
268  }
269 
270  // Add a final residual block to enforce consistancy across the entire grid,
271  // i.e. that the expected number of sources matches the specified number.
272  problem.AddResidualBlock(
273  BeliefRegularization::Create(num_rows_, num_cols_, num_sources_,
274  regularizer_ * viewed_.size()),
275  NULL, /* squared loss */
276  belief_.data());
277 
278  // Set bounds constraints. Each voxel's belief should be a probability
279  // between 0 and 1.
280  for (int ii = 0; ii < num_rows_ * num_cols_; ii++) {
281  problem.SetParameterLowerBound(belief_.data(), ii, 0.0);
282  problem.SetParameterUpperBound(belief_.data(), ii, 1.0);
283  }
284 
285  // Set up solver options.
286  ceres::Solver::Summary summary;
287  ceres::Solver::Options options;
288  options.linear_solver_type = ceres::DENSE_QR;
289  options.function_tolerance = 1e-16;
290  options.gradient_tolerance = 1e-16;
291  options.trust_region_strategy_type = ceres::LEVENBERG_MARQUARDT;
292 
293  // Solve and return.
294  ceres::Solve(options, &problem, &summary);
295 
296  return summary.IsSolutionUsable();
297  }
