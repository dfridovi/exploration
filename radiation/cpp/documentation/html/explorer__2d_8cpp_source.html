76 Explorer2D::Explorer2D(unsigned int num_rows, unsigned int num_cols,
77  unsigned int num_sources, double regularizer,
78  double fov)
79  : map_(num_rows, num_cols, num_sources, regularizer),
80  pose_(0.0, 0.0, 0.0),
81  fov_(fov) {
82  // Set up a random number generator.
83  std::random_device rd;
84  std::default_random_engine rng(rd());
85  std::uniform_int_distribution<unsigned int> unif_rows(0, num_rows - 1);
86  std::uniform_int_distribution<unsigned int> unif_cols(0, num_cols - 1);
87  std::uniform_real_distribution<double> unif_angle(0.0, 2.0 * M_PI);
88 
89  // Choose random sources.
90  for (unsigned int ii = 0; ii < num_sources; ii++) {
91  const Source2D source(unif_rows(rng), unif_cols(rng));
92  sources_.push_back(source);
93  }
94 
95  // Choose a random initial pose.
96  pose_ = GridPose2D(unif_rows(rng), unif_cols(rng), unif_angle(rng));
97 }

99 // Take a step along the given trajectory. Return resulting entropy.
100 double Explorer2D::TakeStep(const std::vector<GridPose2D>& trajectory) {
101  CHECK(trajectory.size() > 0);
102 
103  // Update list of past poses.
104  past_poses_.push_back(pose_);
105 
106  // Update the current pose.
107  pose_ = trajectory[0];
108 
109  // Update the map, and return entropy.
110  const Sensor2D sensor(pose_, fov_);
111  map_.Update(sensor, sources_, true);
112 
113  return map_.Entropy();
114 }

119 // Visualize the current belief state.
120 void Explorer2D::Visualize() const {
121  glClear(GL_COLOR_BUFFER_BIT);
122 
123  // Display each grid cell as a GL_QUAD centered at the appropriate location,
124  // with a small 'epsilon' fudge factor between cells.
125  const Eigen::MatrixXd belief = map_.GetImmutableBelief();
126  const GLfloat kEpsilon = 0.02;
127 
128  glBegin(GL_QUADS);
129  for (unsigned int ii = 0; ii < map_.GetNumRows(); ii++) {
130  for (unsigned int jj = 0; jj < map_.GetNumCols(); jj++) {
131  glColor3f(static_cast<GLfloat>(belief(ii, jj)),
132  static_cast<GLfloat>(belief(ii, jj)),
133  static_cast<GLfloat>(belief(ii, jj)));
134 
135  // Bottom left, bottom right, top right, top left.
136  glVertex2f(static_cast<GLfloat>(ii) + kEpsilon,
137  static_cast<GLfloat>(jj) + kEpsilon);
138  glVertex2f(static_cast<GLfloat>(ii) + 1.0 - kEpsilon,
139  static_cast<GLfloat>(jj) + kEpsilon);
140  glVertex2f(static_cast<GLfloat>(ii) + 1.0 - kEpsilon,
141  static_cast<GLfloat>(jj) + 1.0 - kEpsilon);
142  glVertex2f(static_cast<GLfloat>(ii) + kEpsilon,
143  static_cast<GLfloat>(jj) + 1.0 - kEpsilon);
144  }
145  }
146  glEnd();
147 
148  const GLfloat robot_x = static_cast<GLfloat>(pose_.GetX());
149  const GLfloat robot_y = static_cast<GLfloat>(pose_.GetY());
150  const GLfloat robot_a = static_cast<GLfloat>(pose_.GetAngle());
151  const unsigned int kNumVertices = 100;
152 
153  // Display the field of view as a triangle fan.
154  const GLfloat kFovRadius =
155  sqrt(static_cast<GLfloat>(map_.GetNumRows() * map_.GetNumRows() +
156  map_.GetNumCols() * map_.GetNumCols()));
157 
158  glBegin(GL_TRIANGLE_FAN);
159  glColor4f(0.0, 0.2, 0.8, 0.2);
160  glVertex2f(robot_x, robot_y);
161  for (unsigned int ii = 0; ii <= kNumVertices; ii++) {
162  const GLfloat angle = robot_a + fov_ *
163  (-0.5 + static_cast<GLfloat>(ii) / static_cast<GLfloat>(kNumVertices));
164  glVertex2f(robot_x + kFovRadius * cos(angle),
165  robot_y + kFovRadius * sin(angle));
166  }
167  glEnd();
168 
169  // Display a circle at the robot's current position. No circle primitive, so
170  // use a polygon with a bunch of vertices.
171  const GLfloat kRobotRadius = 0.5;
172 
173  glBegin(GL_POLYGON);
174  glColor4f(0.0, 0.8, 0.2, 0.5);
175  for (unsigned int ii = 0; ii < kNumVertices; ii++) {
176  const GLfloat angle = 2.0 * M_PI *
177  static_cast<GLfloat>(ii) / static_cast<GLfloat>(kNumVertices);
178  glVertex2f(robot_x + kRobotRadius * cos(angle),
179  robot_y + kRobotRadius * sin(angle));
180  }
181  glEnd();
182 
183  // Display a circle at the location of each source.
184  const GLfloat kSourceRadius = 0.2;
185 
186  for (const auto& source : sources_) {
187  glBegin(GL_POLYGON);
188  glColor4f(0.8, 0.0, 0.2, 0.5);
189 
190  for (unsigned int ii = 0; ii < kNumVertices; ii++) {
191  const GLfloat angle = 2.0 * M_PI *
192  static_cast<GLfloat>(ii) / static_cast<GLfloat>(kNumVertices);
193  glVertex2f(source.GetX() + kSourceRadius * cos(angle),
194  source.GetY() + kSourceRadius * sin(angle));
195  }
196  glEnd();
197  }
198 }
